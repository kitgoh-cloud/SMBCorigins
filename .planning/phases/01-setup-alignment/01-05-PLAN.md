```bash
# Existence + non-empty checks
for f in docs/ORIGIN_PRODUCT_BRIEF.html docs/ORIGIN_DESIGN.md docs/ORIGIN_JOURNEY_DOC.html docs/ORIGIN_BUILD_PROMPT.md; do
  if [ -s "$f" ]; then
    echo "OK    $f ($(wc -c < "$f") bytes)"
  else
    echo "FAIL  $f (missing or empty)"
  fi
done

# HTML well-formedness sanity check (looking for closing </html> tag — a forgiving check, not a strict validator)
for f in docs/ORIGIN_PRODUCT_BRIEF.html docs/ORIGIN_JOURNEY_DOC.html; do
  if grep -qi "</html>" "$f"; then
    echo "OK    $f has closing </html>"
  else
    echo "WARN  $f does not contain closing </html> tag"
  fi
done

# Markdown sanity check (presence of at least one H1 heading)
for f in docs/ORIGIN_DESIGN.md docs/ORIGIN_BUILD_PROMPT.md; do
  if grep -qE "^# " "$f"; then
    echo "OK    $f has top-level heading"
  else
    echo "WARN  $f has no top-level # heading"
  fi
done

# CLAUDE.md cross-reference: confirm CLAUDE.md still names all four
for name in ORIGIN_PRODUCT_BRIEF.html ORIGIN_DESIGN.md ORIGIN_JOURNEY_DOC.html ORIGIN_BUILD_PROMPT.md; do
  if grep -q "$name" CLAUDE.md; then
    echo "OK    CLAUDE.md references $name"
  else
    echo "FAIL  CLAUDE.md does NOT reference $name"
  fi
done
```

Then capture all output verbatim into `.planning/phases/01-setup-alignment/01-05-SUMMARY.md` so the Phase 1 closure PR can cite it.
